#include <stdio.h>#include <unistd.h>
    int main (){
            int i,j;
            for (j=0;j<23;j++){
                    printf("\n");
            }
            printf("[   >\t\t\t\t\t\t\t\t   ]");
            for (j=0;j<64;j++)
                    printf("\b");
            for (i=0;i<61;i++){
                    printf("\b");
                    for (j=i;j<i+1;j++)
                            printf("=");
                    printf(">");
                    for (j=0;j<64-i;j++)
                            printf(" ");
                    printf("] %d\%",i);
                    for (j=0;j<67-i;j++)
                            printf("\b");
                    for (j=i;j/10!=0;j=j/10)
                            printf("\b");
                    fflush(stdout);
                    sleep (1);
            }
            printf("\n");
            return 0;
    }
